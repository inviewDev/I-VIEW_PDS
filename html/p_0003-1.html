        type: 'bar',  // 차트 유형
        height: 550, // 차트 높이
        stacked: true, // 누적 차트 (동일한 차트에서만 적용 가능)

        opacity: 1
      }
    };
    // 차트
    var chart = new ApexCharts(document.querySelector("#barChart"), options);
