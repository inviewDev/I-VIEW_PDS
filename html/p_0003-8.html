    JS

    // 차트 옵션
    var options = {
      series: [{ // 차트 데이터
           name: "Current Week",
           data: [10, 20, 15, 25, 20, 30, 20]
       }, {
           name: "Previous Week",
           data: [0, 15, 10, 30, 15, 35, 25]
       }],
      colors:["#727cf5", "#0acf97"], // 차트 기본 색상 ( 데이터가 하나라고 해도 배열로 해야함 )
      chart: {
        width:'100%',
        height: 500,
        type: 'line',
        foreColor: '#008ffb', // 차트의 텍스트 색상
        dropShadow: { // 차트 라인 그림자
           enabled: 1,
           opacity: .2,
           blur: 7,
           left: -7,
           top: 7
       },
       toolbar:{ // 차트 다운로드 기능
         show: false
       },
        zoom: { // 차트 구간 확대/축소
          enabled: false
        },
        animations: { // 로딩시 애니메이션
            enabled: true,
            easing: 'linear'
        }
      },
      dataLabels: { // 데이터라벨
        enabled: true,
        formatter: function(e) { // 라벨값에 대한 사용자 
          return e + "k"
        },
        background: { // 데이터 라벨 영역 공간 만듬
          enabled: true,
          padding: 10
        }
      },
      legend:{ // 범례
        labels:{
          useSeriesColors: true // 기본 색상 적용 유무
        }
      },
      stroke: { // 라인차트 형태
        curve: "smooth", // straight : 직선, smooth : 곡선, stepline : 계단형태
        width: 4
      },
      xaxis: { // x축
         type: "string",
         categories: ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"],
         tooltip: {
             enabled: false
         },
         axisBorder: { // x축 가로 border 
             show: false
         }
     },
      yaxis:{ // y축
        labels:{
          formatter: function(e) { // y축 값에 대한 사용자 정의
            return e + "k" // k 단위 표시
          },
          offsetX: -10
        }
      }
    };
    // 차트
    var chart = new ApexCharts(document.querySelector("#lineChart"), options);
    chart.render();
